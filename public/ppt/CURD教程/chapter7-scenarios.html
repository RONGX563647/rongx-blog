🔍 场景1：搜索框与表格联动查询

📝 场景描述

用户输入搜索关键词后，自动触发表格查询，支持多条件组合搜索，输入时防抖避免频繁请求。

前端实现（Vue3）

<template>
  <el-form :model="queryParams" :inline="true">
    <el-form-item label="关键词">
      <el-input v-model="queryParams.keyword" placeholder="搜索用户名/邮箱" clearable @input="handleSearch" />
    </el-form-item>
    <el-form-item label="状态">
      <el-select v-model="queryParams.status" placeholder="全部" clearable @change="handleSearch">
        <el-option label="启用" :value="1" />
        <el-option label="禁用" :value="0" />
      </el-select>
    </el-form-item>
    <el-form-item label="日期范围">
      <el-date-picker v-model="dateRange" type="daterange" @change="handleSearch" />
    </el-form-item>
  </el-form>
</template>

<script setup>
import { ref, watch } from 'vue'

const queryParams = ref({ keyword: '', status: undefined })
const dateRange = ref([])
let searchTimer = null

// 防抖搜索
const handleSearch = () => {
  if (searchTimer) clearTimeout(searchTimer)
  searchTimer = setTimeout(() => {
    // 组装日期参数
    if (dateRange.value && dateRange.value.length === 2) {
      queryParams.value.startDate = dateRange.value[0]
      queryParams.value.endDate = dateRange.value[1]
    } else {
      delete queryParams.value.startDate
      delete queryParams.value.endDate
    }
    fetchData()
  }, 300)
}

// 监听日期变化
watch(dateRange, () => {
  handleSearch()
})
</script>

后端实现（MyBatis Plus条件构造器）

// Controller
@GetMapping
public Result<PageResult<User>> list(UserQueryDTO query) {

}

// Service - 多条件组合查询
public PageResult<User> queryPage(UserQueryDTO query) {
  Page<User> page = new Page<>(query.getPageNum(), query.getPageSize());
  
  LambdaQueryWrapper<User> wrapper = new LambdaQueryWrapper<>();
  // 关键词搜索（用户名 OR 邮箱）
  if (StringUtils.hasText(query.getKeyword())) {
    wrapper.and(w -> w.like(User::getUsername, query.getKeyword())
        .or().like(User::getEmail, query.getKeyword()));
  }
  // 状态筛选
  wrapper.eq(query.getStatus() != null, User::getStatus, query.getStatus());
  // 日期范围
  wrapper.ge(query.getStartDate() != null, User::getCreateTime, query.getStartDate());
  wrapper.le(query.getEndDate() != null, User::getCreateTime, query.getEndDate());
  
  wrapper.orderByDesc(User::getCreateTime);
  Page<User> result = this.page(page, wrapper);
  return new PageResult<>(result.getRecords(), result.getTotal());
}

🔄 场景2：表格内状态快速切换

📝 场景描述

在表格中直接通过Switch组件切换用户状态，切换失败时自动回滚，切换前二次确认。

前端实现

<template>
  <el-table :data="userList">
    <el-table-column label="状态" width="100">
      <template #default="{ row }">
        <el-switch
          v-model="row.status"
          :active-value="1"
          :inactive-value="0"
          :loading="row.switchLoading"
          @change="handleStatusChange(row)"
        />
      </template>
    </el-table-column>
  </el-table>
</template>

<script setup>
const handleStatusChange = async (row) => {
  const oldStatus = row.status === 1 ? 0 : 1 // 记录原状态
  
  try {
    row.switchLoading = true
    await ElMessageBox.confirm(
      `确认${row.status === 1 ? '启用' : '禁用'}用户"${row.username}"吗？`,
      '提示',
      { type: 'warning' }
    )
    
    await updateUserStatus(row.id, row.status)
    ElMessage.success('状态更新成功')
  } catch (error) {
    // 失败回滚状态
    row.status = oldStatus
    if (error !== 'cancel') {
      ElMessage.error('状态更新失败')
    }
  } finally {
    row.switchLoading = false
  }
}
</script>

后端实现

@PutMapping("/{id}/status")
public Result<Void> updateStatus(@PathVariable Long id, @RequestParam Integer status) {
  // 参数校验
  if (status != 0 && status != 1) {
    return Result.error(400, "状态值无效");
  }
  
  User user = userService.getById(id);
  if (user == null) {
    return Result.error(404, "用户不存在");
  }
  
  // 不允许禁用超级管理员
  if (user.isAdmin() && status == 0) {
    return Result.error(403, "不能禁用超级管理员");
  }
  
  user.setStatus(status);
  userService.updateById(user);
  return Result.success();
}

后端实现（EasyExcel）

@PostMapping("/import")
public Result<ImportResult> importUsers(@RequestParam("file") MultipartFile file) {
  try {
    List<UserImportDTO> dataList = EasyExcel.read(file.getInputStream())
        head(UserImportDTO.class)
        sheet()
        doReadSync();
    
    if (dataList.size() > 1000) {
      return Result.error("单次最多导入1000条");
    }
    
    ImportResult result = userService.batchImport(dataList);
    return Result.success(result);
  } catch (Exception e) {
    return Result.error("文件解析失败：" + e.getMessage());
  }
}

// Service批量导入
@Transactional
public ImportResult batchImport(List<UserImportDTO> dataList) {
  int successCount = 0;
  int failCount = 0;
  List<String> errors = new ArrayList<>();
  
  // 查询已存在的用户名
  List<String> existingUsernames = this.list()
    .stream().map(User::getUsername).collect(Collectors.toList());
  
  List<User> toInsert = new ArrayList<>();
  for (int i = 0; i < dataList.size(); i++) {
    UserImportDTO dto = dataList.get(i);
    int rowNum = i + 2; // Excel行号
    
    // 数据验证
    if (existingUsernames.contains(dto.getUsername())) {
      errors.add("第" + rowNum + "行：用户名\"" + dto.getUsername() + "\"已存在");
      failCount++;
      continue;
    }
    
    User user = new User();
    user.setUsername(dto.getUsername());
    user.setEmail(dto.getEmail());
    user.setPhone(dto.getPhone());
    user.setStatus(1);
    toInsert.add(user);
    successCount++;
  }
  
  // 批量插入
  if (!toInsert.isEmpty()) {
    this.saveBatch(toInsert);
  }
  
  return new ImportResult(successCount, failCount, errors);
}

@PostMapping("/upload/avatar")
public Result<FileInfo> uploadAvatar(@RequestParam("file") MultipartFile file) {
  // 验证文件类型
  String contentType = file.getContentType();
  if (!contentType.startsWith("image/")) {
    return Result.error("只允许上传图片文件");
  }
  
  // 验证文件大小（2MB）
  if (file.getSize() > 2 * 1024 * 1024) {
    return Result.error("文件大小不能超过2MB");
  }
  
  // 生成文件名
  String originalFilename = file.getOriginalFilename();
  String extension = originalFilename.substring(originalFilename.lastIndexOf("."));
  String fileName = UUID.randomUUID().toString() + extension;
  String filePath = "/uploads/avatars/" + fileName;
  
  // 保存文件（实际应使用OSS等云存储）
  Path path = Paths.get(uploadDir + filePath);
  Files.createDirectories(path.getParent());
  file.transferTo(path);
  
  FileInfo fileInfo = new FileInfo();
  fileInfo.setUrl("/api/files" + filePath);
  return Result.success(fileInfo);
}

📚 场景8：数据字典动态下拉框

📝 场景描述

从后端获取字典数据，动态渲染下拉选项，支持字典缓存减少请求。

// 前端：字典Hook
import { ref, onMounted } from 'vue'

const dictCache = new Map()

export function useDict(dictType) {
  const options = ref([])
  const loading = ref(false)
  
  const loadDict = async () => {
    // 检查缓存
    if (dictCache.has(dictType)) {
      options.value = dictCache.get(dictType)
      return
    }
    
    loading.value = true
    try {
      const res = await getDictData(dictType)
      options.value = res.data
      dictCache.set(dictType, res.data) // 缓存
    } finally {
      loading.value = false
    }
  }
  
  onMounted(loadDict)
  return { options, loading }
}

// 使用示例
const { options: statusOptions } = useDict('user_status')

// 模板中使用
<el-select v-model="form.status">
  <el-option v-for="item in statusOptions" :key="item.value" :label="item.label" :value="item.value" />
</el-select>

🧠 费曼自测：你能解释给同学听吗？

Q1：搜索框防抖为什么要延迟300毫秒？

A：想象你在打字——每按一个键就发一次请求，服务器会崩溃。300毫秒是人打字间隔的上限，等用户停下来0.3秒后再发送请求，是最佳平衡点。

Q2：批量删除为什么要先勾选再删除？

A：就像在文件管理器里操作——按住Ctrl一个一个选，确认好再右键删除。单独的删除按钮和批量删除按钮是为了防止误操作——单独删除就删一个，批量删除要确认N个。

Q3：Monaco编辑器和普通textarea有什么区别？

A：textarea就像WordPad（简陋），Monaco就像VS Code（专业）。后者支持语法高亮、代码补全、多光标编辑——适用于需要用户写代码或配置的场景。
