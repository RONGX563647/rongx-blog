🔄 场景1：审批流程引擎（请假审批）

📝 业务背景

企业内部的请假申请需要经过多级审批：员工提交 → 直属领导审批 → 部门经理审批 → HR备案。不同天数对应不同审批级别，需要实现动态流程、审批记录追踪、审批状态实时查看等功能。

// LeaveApplicationService.java @Service @RequiredArgsConstructor public class LeaveApplicationService extends ServiceImpl<LeaveApplicationMapper, LeaveApplication> { private final ApprovalRecordService approvalRecordService; private final UserService userService; /** * 提交请假申请 */ @Transactional(rollbackFor = Exception.class) public void submitApplication(LeaveApplicationDTO dto, Long userId) { // 1. 创建申请 LeaveApplication application = new LeaveApplication(); application.setUserId(userId); application.setLeaveType(dto.getLeaveType()); application.setStartDate(dto.getStartDate()); application.setEndDate(dto.getEndDate()); application.setDays(dto.getDays()); application.setReason(dto.getReason()); application.setStatus("pending_first_approval"); // 2. 根据天数确定第一级审批人 Long firstApproverId = getApproverByDays(userId, dto.getDays(), 1); application.setCurrentApproverId(firstApproverId); this.save(application); // 3. 记录提交日志 approvalRecordService.saveRecord( application.getId(), userId, "submit", "提交请假申请" ); // 4. 发送通知给审批人（消息队列/WebSocket） notifyApprover(firstApproverId, application); } /** * 审批处理 */ @Transactional(rollbackFor = Exception.class) public void approveApplication(Long applicationId, Long approverId, String action, String comment) { LeaveApplication application = this.getById(applicationId); if (application == null) { throw new BusinessException("申请不存在"); } // 验证审批权限 if (!application.getCurrentApproverId().equals(approverId)) { throw new BusinessException("无审批权限"); } // 记录审批 approvalRecordService.saveRecord(applicationId, approverId, action, comment); if ("approve".equals(action)) { // 审批通过 handleApprovalPass(application); } else if ("reject".equals(action)) { // 驳回 application.setStatus("rejected"); this.updateById(application); notifyApplicant(application.getUserId(), "您的请假申请已被驳回"); } } /** * 审批通过后的流转逻辑 */ private void handleApprovalPass(LeaveApplication application) { int currentLevel = getCurrentApprovalLevel(application); int totalLevels = getTotalApprovalLevels(application.getDays()); if (currentLevel >= totalLevels) { // 所有审批完成 application.setStatus("approved"); application.setCurrentApproverId(null); } else { // 流转到下一级审批人 int nextLevel = currentLevel + 1; Long nextApproverId = getApproverByDays( application.getUserId(), application.getDays(), nextLevel ); application.setStatus("pending_" + getLevelName(nextLevel)); application.setCurrentApproverId(nextApproverId); // 通知下一级审批人 notifyApprover(nextApproverId, application); } this.updateById(application); } /** * 根据天数获取对应级别的审批人 */ private Long getApproverByDays(Long applicantId, int days, int level) { User applicant = userService.getById(applicantId); return switch (level) { case 1 -> applicant.getManagerId(); // 直属领导 case 2 -> applicant.getDepartmentManagerId(); // 部门经理 case 3 -> getGeneralManagerId(); // 总经理 default -> throw new BusinessException("审批级别不存在"); }; } /** * 获取总审批级别数 */ private int getTotalApprovalLevels(int days) { if (days <= 3) return 1; if (days <= 7) return 2; return 3; } }

// LeaveController.java
@RestController
@RequestMapping("/api/leave")
@RequiredArgsConstructor
public class LeaveController {

    private final LeaveApplicationService leaveService;

    /** 提交请假申请 */
    @PostMapping
    @AuditLog(value = "提交请假申请", module = "审批管理")
    public Result<Void> submit(@RequestBody @Valid LeaveApplicationDTO dto) {
        leaveService.submitApplication(dto, SecurityUtils.getCurrentUserId());
        return Result.success();
    }

    /** 审批（通过/驳回） */
    @PutMapping("/{id}/approve")
    @AuditLog(value = "请假审批", module = "审批管理")
    public Result<Void> approve(@PathVariable Long id, @RequestBody @Valid ApproveDTO dto) {
        leaveService.approveApplication(id, SecurityUtils.getCurrentUserId(), dto.getAction(), dto.getComment());
        return Result.success();
    }

    /** 撤销申请（只有"待审批"状态可以撤销） */
    @PutMapping("/{id}/cancel")
    @AuditLog(value = "撤销请假申请", module = "审批管理")
    public Result<Void> cancel(@PathVariable Long id) {
        leaveService.cancelApplication(id, SecurityUtils.getCurrentUserId());
        return Result.success();
    }

    /** 查询我的请假记录 */
    @GetMapping("/my")
    public Result<PageResult<LeaveApplicationVO>> queryMyLeaves(LeaveQueryDTO query) {
        query.setUserId(SecurityUtils.getCurrentUserId());
        return Result.success(leaveService.queryPage(query));
    }
}

// 撤销逻辑（Service层）
@Transactional(rollbackFor = Exception.class)
public void cancelApplication(Long applicationId, Long userId) {
    LeaveApplication application = this.getById(applicationId);
    if (application == null) {
        throw new BusinessException("申请不存在");
    }
    if (!application.getUserId().equals(userId)) {
        throw new BusinessException("只能撤销自己的申请");
    }
    if (!"pending_first_approval".equals(application.getStatus())) {
        throw new BusinessException("只有待审批状态的申请才能撤销");
    }

    application.setStatus("cancelled");
    this.updateById(application);
    approvalRecordService.saveRecord(applicationId, userId, "cancel", "申请人主动撤销");
}

💡 审批流程最佳实践

1. 状态机设计：使用明确的状态值，避免状态混乱
2. 权限校验：每次审批前验证当前审批人身份
3. 事务管理：审批操作和状态更新必须在同一事务中
4. 通知机制：审批流转后及时通知相关人员
5. 审批记录：保留完整的审批历史，支持追溯
6. 流程配置化：将审批规则配置化，支持动态调整
7. 撤销机制：申请人可在审批完成前主动撤销，已审批的申请不可撤销

// 1. 自定义注解 @Target(ElementType.METHOD) @Retention(RetentionPolicy.RUNTIME) public @interface DataScope { /** * 表别名（默认""） */ String tableAlias() default ""; /** * 部门字段名（默认"department_id"） */ String deptField() default "department_id"; /** * 用户字段名（默认"user_id"） */ String userField() default "user_id"; } // 2. MyBatis拦截器实现数据权限过滤 @Intercepts({ @Signature(type = Executor.class, method = "query", args = {MappedStatement.class, Object.class, RowBounds.class, ResultHandler.class}) }) @Component @RequiredArgsConstructor public class DataScopeInterceptor implements Interceptor { private final UserService userService; @Override public Object intercept(Invocation invocation) throws Throwable { MappedStatement ms = (MappedStatement) invocation.getArgs()[0]; Object parameter = invocation.getArgs()[1]; // 获取方法上的@DataScope注解 DataScope dataScope = getDataScopeAnnotation(ms); if (dataScope == null) { return invocation.proceed(); } // 获取当前用户 LoginUser currentUser = SecurityUtils.getCurrentUser(); if (currentUser == null || currentUser.isAdmin()) { return invocation.proceed(); // 管理员无限制 } // 获取用户数据权限范围 DataScopeInfo scopeInfo = getUserDataScope(currentUser.getId()); // 构建权限SQL String dataScopeSql = buildDataScopeSql(dataScope, scopeInfo); // 修改原始SQL，追加WHERE条件 BoundSql boundSql = ms.getBoundSql(parameter); String originalSql = boundSql.getSql(); String newSql = originalSql + " " + dataScopeSql; // 重新设置SQL MetaObject metaObject = SystemMetaObject.forObject(boundSql); metaObject.setValue("sql", newSql); return invocation.proceed(); } /** * 构建数据权限SQL片段 */ private String buildDataScopeSql(DataScope annotation, DataScopeInfo scopeInfo) { String tableAlias = annotation.tableAlias(); String deptField = annotation.deptField(); String userField = annotation.userField(); String alias = StringUtils.hasText(tableAlias) ? tableAlias + "." : ""; return switch (scopeInfo.getScopeType()) { case "all" -> ""; // 全部数据权限 case "dept" -> String.format("AND %s%s = %d", alias, deptField, scopeInfo.getDeptId()); case "dept_and_children" -> { List<Long> deptIds = getChildrenDeptIds(scopeInfo.getDeptId()); deptIds.add(scopeInfo.getDeptId()); yield String.format("AND %s%s IN (%s)", alias, deptField, StringUtils.join(deptIds, ",")); } case "self" -> String.format("AND %s%s = %d", alias, userField, scopeInfo.getUserId()); case "custom" -> String.format("AND %s%s IN (%s)", alias, deptField, scopeInfo.getCustomDeptIds()); default -> "AND 1=0"; // 无权限 }; } /** * 获取用户的数据权限范围 */ private DataScopeInfo getUserDataScope(Long userId) { User user = userService.getById(userId); DataScopeConfig config = dataScopeConfigMapper.getByRoleCode(user.getRoleCode()); DataScopeInfo info = new DataScopeInfo(); info.setUserId(userId); info.setDeptId(user.getDepartmentId()); info.setScopeType(config.getScopeType()); info.setCustomDeptIds(config.getCustomDeptIds()); return info; } } // 3. 在Mapper方法上使用 public interface UserMapper extends BaseMapper<User> { @DataScope(tableAlias = "u", deptField = "department_id", userField = "user_id") List<User> selectUserList(@Param("query") UserQueryDTO query); }

⚠️ 数据权限注意事项

1. 越权防护：后端必须做数据权限校验，前端只是UI展示控制
2. 性能影响：数据权限SQL可能影响查询性能，需要优化索引
3. 跨表查询：多表JOIN时需要正确处理表别名
4. 缓存问题：权限变更时需要清理相关缓存

// 1. Service层实现（避免N+1查询） @Service @RequiredArgsConstructor public class OrderService { private final OrderMapper orderMapper; private final OrderItemMapper orderItemMapper; private final UserMapper userMapper; private final LogisticsMapper logisticsMapper; /** * 分页查询订单列表（优化版） */ public PageResult<OrderVO> queryOrderPage(OrderQueryDTO query) { // 1. 分页查询订单主表（只查ID，避免JOIN影响分页） Page<Order> orderPage = new Page<>(query.getPageNum(), query.getPageSize()); LambdaQueryWrapper<Order> wrapper = buildQueryWrapper(query); Page<Order> result = orderMapper.selectPage(orderPage, wrapper); if (result.getRecords().isEmpty()) { return new PageResult<>(new ArrayList<>(), 0); } // 2. 批量查询关联数据（避免N+1问题） List<Long> orderIds = result.getRecords().stream() .map(Order::getId) .collect(Collectors.toList()); List<Long> userIds = result.getRecords().stream() .map(Order::getUserId) .distinct() .collect(Collectors.toList()); // 批量查询用户信息 Map<Long, User> userMap = userMapper.selectBatchIds(userIds).stream() .collect(Collectors.toMap(User::getId, u -> u)); // 批量查询订单项 List<OrderItem> allItems = orderItemMapper.selectList( new LambdaQueryWrapper<OrderItem>().in(OrderItem::getOrderId, orderIds) ); Map<Long, List<OrderItem>> itemsMap = allItems.stream() .collect(Collectors.groupingBy(OrderItem::getOrderId)); // 批量查询物流信息 Map<Long, Logistics> logisticsMap = logisticsMapper.selectList( new LambdaQueryWrapper<Logistics>().in(Logistics::getOrderId, orderIds) ).stream().collect(Collectors.toMap(Logistics::getOrderId, l -> l)); // 3. 组装VO List<OrderVO> orderVOList = result.getRecords().stream() .map(order -> { OrderVO vo = new OrderVO(); BeanUtils.copyProperties(order, vo); // 设置用户信息 User user = userMap.get(order.getUserId()); if (user != null) { vo.setUsername(user.getUsername()); vo.setUserPhone(user.getPhone()); } // 设置订单项 List<OrderItem> items = itemsMap.getOrDefault(order.getId(), new ArrayList<>()); vo.setItems(items.stream().map(item -> { OrderItemVO itemVO = new OrderItemVO(); BeanUtils.copyProperties(item, itemVO); return itemVO; }).collect(Collectors.toList())); // 设置物流信息 Logistics logistics = logisticsMap.get(order.getId()); if (logistics != null) { vo.setLogistics(new LogisticsVO(logistics)); } return vo; }) .collect(Collectors.toList()); return new PageResult<>(orderVOList, result.getTotal()); } private LambdaQueryWrapper<Order> buildQueryWrapper(OrderQueryDTO query) { LambdaQueryWrapper<Order> wrapper = new LambdaQueryWrapper<>(); // 订单号搜索 if (StringUtils.hasText(query.getOrderNo())) { wrapper.like(Order::getOrderNo, query.getOrderNo()); } // 状态筛选 if (query.getStatus() != null) { wrapper.eq(Order::getStatus, query.getStatus()); } // 用户搜索 if (StringUtils.hasText(query.getUserKeyword())) { // 先查询匹配的用户ID List<Long> userIds = userMapper.selectList( new LambdaQueryWrapper<User>() .like(User::getUsername, query.getUserKeyword()) .or() .like(User::getPhone, query.getUserKeyword()) ).stream().map(User::getId).collect(Collectors.toList()); if (!userIds.isEmpty()) { wrapper.in(Order::getUserId, userIds); } else { wrapper.eq(Order::getUserId, -1L); // 无匹配结果 } } // 时间范围 wrapper.ge(query.getStartDate() != null, Order::getCreateTime, query.getStartDate()); wrapper.le(query.getEndDate() != null, Order::getCreateTime, query.getEndDate()); wrapper.orderByDesc(Order::getCreateTime); return wrapper; } }
🎯 关键优化点：
1. 分页不JOIN：先分页查主表ID，再批量查关联表，避免JOIN导致分页不准确
2. 批量查询：使用IN查询批量获取关联数据，避免N+1问题
3. 内存组装：在Java内存中组装VO，减少数据库压力
4. Map索引：将关联数据转为Map，提高查找效率

return Boolean.TRUE.equals(success); } /** * 释放锁（Lua脚本保证原子性） */ public void releaseLock(String key, String value) { String script = "if redis.call('get', KEYS[1]) == ARGV[1] then " + "return redis.call('del', KEYS[1]) " + "else return 0 end"; redisTemplate.execute( new DefaultRedisScript<>(script, Long.class), Collections.singletonList(key), value ); } /** * 带重试的锁获取 */ public boolean tryLockWithRetry(String key, String value, long expireSeconds, int retryCount, long retryInterval) { for (int i = 0; i < retryCount; i++) { if (tryLock(key, value, expireSeconds)) { return true; } try { Thread.sleep(retryInterval); } catch (InterruptedException e) { Thread.currentThread().interrupt(); } } return false; } } // 使用示例：秒杀场景 @Service @RequiredArgsConstructor public class SeckillService { private final RedisLock redisLock; private final ProductService productService; private final StringRedisTemplate redisTemplate; } // 2. 检查用户是否已购买（幂等性） String purchaseKey = "seckill:purchase:" + productId + ":" + userId; if (Boolean.TRUE.equals(redisTemplate.hasKey(purchaseKey))) { throw new BusinessException("您已购买过该商品"); } // 3. 扣减库存 boolean success = productService.deductStock(productId, 1); } } }

⚠️ 并发控制注意事项

1. 死锁预防：避免嵌套锁，按固定顺序获取锁
2. 锁超时：所有锁必须设置超时时间，防止死锁
3. 幂等性：关键操作必须保证幂等，重试不会产生副作用
4. 性能权衡：悲观锁性能差但安全，乐观锁性能好但需要重试机制
5. Redis锁可靠性：Redis单点故障可能导致锁失效，生产环境建议使用Redisson

🛑 场景12：Spring Boot优雅停机

# application.yml server: shutdown: graceful # 优雅停机 spring: lifecycle: timeout-per-shutdown-phase: 30s # 等待时间 # Spring Boot 2.3+ 默认支持优雅停机 # 收到SIGTERM信号后： # 1. 不再接受新请求 # 2. 等待已有请求处理完（最多30s） # 3. 销毁Bean，释放资源 // 自定义关闭回调 @Component public class GracefulShutdown implements DisposableBean { @Override public void destroy() { log.info("开始优雅关闭..."); // 1. 通知注册中心下线 nacosServiceRegistry.deregister(); // 2. 等待任务队列处理完 executorService.shutdown(); executorService.awaitTermination(30, TimeUnit.SECONDS); // 3. 刷新缓冲区 logBuffer.flush(); log.info("优雅关闭完成"); } } // docker-compose中的stop_grace_period services: backend: stop_grace_period: 30s # 给30秒优雅停机时间

🧠 费曼自测：你能解释给同学听吗？

Q1：乐观锁到底是什么？（用抢车位比喻）

A：你开车去停车场抢车位，先假设没人跟你抢（乐观），开到车位前发现已经被占了，那就换一个（版本号不匹配，重试）。悲观锁则相反——你直接锁住整个停车场不让别人进——效率低但不会冲突。

Q2：多租户系统为什么要隔离数据？

A：就像公寓楼——每个租户只能进自己的房间。你绝对不能看到邻居的账单。tenant_id就是房间钥匙，MyBatis拦截器自动帮你加上"where tenant_id=?"条件。

Q3：审计日志和普通日志有什么区别？

A：普通日志 = 监控摄像头（出事了翻录像）；审计日志 = 签字画押（每笔操作必须留下不可篡改的证据）。审计日志通常独立存储（甚至存ES），用于合规审查。
